"""
GEBCO Bathymetry Client - REAL IMPLEMENTATION with LOCAL GRID SUPPORT
Priority: Local Grid > WMS > Estimation
"""

import os
import requests
import numpy as np
from typing import Dict, Optional
from datetime import datetime
import logging
from io import BytesIO
from PIL import Image

logger = logging.getLogger(__name__)

# Try to import local grid reader
try:
    from .gebco_local_reader import GEBCOLocalReader
    LOCAL_GRID_AVAILABLE = True
except ImportError:
    LOCAL_GRID_AVAILABLE = False


class GEBCOClient:
    """GEBCO Bathymetry client with local grid support"""

    WMS_URL = "https://www.gebco.net/data_and_products/gebco_web_services/web_map_service/mapserv"

    @staticmethod
    def fetch_bathymetry(lat_min: float, lat_max: float, lng_min: float, lng_max: float,
                        resolution: int = 100, use_cache: bool = True) -> Optional[Dict]:
        """
        Fetch bathymetry data - tries local grid first, then WMS, then estimation
        """
        no_mock = os.environ.get('NO_MOCK_DATA', 'False').lower() == 'true'

        # PRIORITY 1: Local GEBCO grid (exact depths)
        if LOCAL_GRID_AVAILABLE and GEBCOLocalReader.is_available():
            try:
                logger.info("Using local GEBCO grid")
                return GEBCOLocalReader.get_depth_grid(lat_min, lat_max, lng_min, lng_max, resolution)
            except Exception as e:
                logger.warning(f"Local grid failed: {e}, trying WMS")

        # PRIORITY 2: GEBCO WMS
        try:
            logger.info("Fetching GEBCO bathymetry via WMS")
            params = {
                'service': 'WMS',
                'version': '1.3.0',
                'request': 'GetMap',
                'layers': 'GEBCO_LATEST',
                'crs': 'EPSG:4326',
                'bbox': f'{lat_min},{lng_min},{lat_max},{lng_max}',
                'width': str(resolution),
                'height': str(resolution),
                'format': 'image/png'
            }

            response = requests.get(GEBCOClient.WMS_URL, params=params, timeout=30)
            response.raise_for_status()

            img = Image.open(BytesIO(response.content))
            img_array = np.array(img)

            if len(img_array.shape) == 3:
                gray = np.mean(img_array[:, :, :3], axis=2)
            else:
                gray = img_array

            max_depth = 6000
            depth_grid = -1 * (max_depth * (1 - gray / 255.0))
            depth_grid[gray > 200] = np.nan

            lats = np.linspace(lat_max, lat_min, resolution)
            lngs = np.linspace(lng_min, lng_max, resolution)

            logger.info(f"[OK] WMS: depth {np.nanmin(depth_grid):.1f}m to {np.nanmax(depth_grid):.1f}m")

            return {
                'date': datetime.utcnow().strftime('%Y-%m-%d'),
                'bathymetry_grid': depth_grid.tolist(),
                'lat': lats.tolist(),
                'lng': lngs.tolist(),
                'source': 'GEBCO_WMS',
                'resolution': f'{resolution}x{resolution}',
                'min_depth': float(np.nanmin(depth_grid)),
                'max_depth': float(np.nanmax(depth_grid)),
                'units': 'meters',
                'note': 'Approximate from WMS'
            }

        except Exception as e:
            logger.warning(f"WMS failed: {e}")
            if no_mock:
                raise
            return GEBCOClient._fallback_depth_estimate(lat_min, lat_max, lng_min, lng_max, resolution)

    @staticmethod
    def get_depth_at_point(lat: float, lng: float) -> float:
        """Get depth at point - uses local grid if available"""
        if LOCAL_GRID_AVAILABLE and GEBCOLocalReader.is_available():
            try:
                return GEBCOLocalReader.get_depth_at_point(lat, lng)
            except:
                pass

        data = GEBCOClient.fetch_bathymetry(lat-0.05, lat+0.05, lng-0.05, lng+0.05, 3)
        if data:
            grid = np.array(data['bathymetry_grid'])
            return float(grid[1, 1]) if not np.isnan(grid[1, 1]) else -100.0
        return -100.0

    @staticmethod
    def _fallback_depth_estimate(lat_min, lat_max, lng_min, lng_max, resolution=100):
        """Geographic depth estimation for Arabian Sea"""
        logger.warning("Using fallback depth estimation")

        lats = np.linspace(lat_max, lat_min, resolution)
        lngs = np.linspace(lng_min, lng_max, resolution)
        LAT, LNG = np.meshgrid(lats, lngs, indexing='ij')

        depth_grid = np.zeros_like(LAT)
        shelf_mask = LNG > 72.0
        depth_grid[shelf_mask] = -50 - np.random.uniform(0, 100, np.sum(shelf_mask))

        slope_mask = (LNG > 71.0) & (LNG <= 72.0)
        slope_dist = (72.0 - LNG[slope_mask]) / 1.0
        depth_grid[slope_mask] = -200 - slope_dist * 1800 - np.random.uniform(0, 200, np.sum(slope_mask))

        deep_mask = LNG <= 71.0
        depth_grid[deep_mask] = -3000 - np.random.uniform(0, 1000, np.sum(deep_mask))

        return {
            'date': datetime.utcnow().strftime('%Y-%m-%d'),
            'bathymetry_grid': depth_grid.tolist(),
            'lat': lats.tolist(),
            'lng': lngs.tolist(),
            'source': 'FALLBACK_ESTIMATE',
            'resolution': f'{resolution}x{resolution}',
            'min_depth': float(np.min(depth_grid)),
            'max_depth': float(np.max(depth_grid)),
            'units': 'meters',
            'note': 'Geographic estimation'
        }
